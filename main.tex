\section{TAD Nat extendido}
\begin{tad}

\tadOtrasOperaciones
\tadOperacion{mod}{nat/n, nat/m}{nat}{}
\tadOperacion{$\bullet$ $/$ $\bullet$}{nat/n, nat/m}{nat}{$\lnot$ m=0?}
\tadOperacion{raiz}{nat/n}{nat}{}

\tadAxiomas[\paratodo{nat}{n,m}]
\tadAxioma{mod(n,m)}{\IF m $>$ n THEN m - n ELSE n - m FI}
\tadAxioma{n $/$ m}{\IF m $>$ n THEN 0 ELSE 1 + (n - m) $/$ (m) FI}

\end{tad}
